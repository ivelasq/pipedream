Fun #rstats trick: want to filter based on another variable? Use `which()`:

library(tidyverse)

dat <-
data.frame(
district_id = c(123L, 123L, 456L, 456L),
cds = c("a", "b", "c", "d")
)

dat %>%
filter(district_id %in% district_id[which(cds == "a")])
